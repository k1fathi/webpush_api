# Initialize the database
./scripts/db_manager.sh init

# Backup data 
./scripts/db_manager.sh backup

# Create a new migration
./scripts/db_manager.sh new-migration add_user_preferences

# Apply migrations
./scripts/db_manager.sh upgrade

# Complete reset (with confirmation)
./scripts/db_manager.sh full-reset




Run database
------------------------------------------
docker exec -it webpush_api-web-1 bash -c "chmod +x ./scripts/db_manager.sh"
docker exec -it webpush_api-web-1 ./scripts/db_manager.sh full-reset


docker exec -it webpush_api-web-1 ./scripts/db_manager.sh init
docker exec -it webpush_api-web-1 ./scripts/db_manager.sh new-migration add_user_preferences

docker exec -it webpush_api-web-1 ./scripts/db_manager.sh sync-and-migrate add_user_preferences
